select srz_pid,ss.SchetSluchID,ds1,s.[year],s.[month],ss.DATE_1,ssa.DATE_Z_2
into #allsluch
	from iesdb.[IES].T_SCHET_SLUCH as ss with(nolock)
	    inner join iesdb.ies.T_SCHET_ZAP as sz with(nolock) on (sz.SchetZapID = ss.SchetZap)
	        inner join iesdb.ies.T_SCHET as s with(nolock) on (s.SchetID=sz.Schet)
	            inner join iesdb.ies.T_SCHET_SLUCH_ACCOMPLISHED as ssa with(nolock) on (ss.SchetSluchAccomplished=ssa.SchetSluchAccomplishedID)
	                where(
		            (s.[year] in (2018,2019,2020) and ssa.oplata=1 and s.type_=693 and isnull(ssa.usl_ok,3)<>4 and (s.FILENAME like 'HS%' or s.FILENAME like 'HCS%' or s.FILENAME like 'D_S%' or s.FILENAME like 'TS%') )
			        or
		            (s.[year] in (2021) and s.[month]<=@month and ssa.oplata=1 and s.type_=693 and isnull(ssa.usl_ok,3)<>4 and (s.FILENAME like 'HM%' or s.FILENAME like 'HCM%' or s.FILENAME like 'D_M%' or s.FILENAME like 'TM%')))
		            and datediff(month,sz.dr,ss.date_1)>=216
		            and (ss.ds1 like 'E10%' or ss.ds1 like 'E11%')


select srz_pid,ss.SchetSluchID,ds1,s.[year],s.[month],ss.DATE_1,ssa.DATE_Z_2
into #allsluchnew
	from iesdb.[IES].T_SCHET_SLUCH as ss with(nolock)
	    inner join iesdb.ies.T_SCHET_ZAP as sz with(nolock) on (sz.SchetZapID = ss.SchetZap)
	        inner join iesdb.ies.T_SCHET as s with(nolock) on (s.SchetID=sz.Schet)
	            inner join iesdb.ies.T_SCHET_SLUCH_ACCOMPLISHED as ssa with(nolock) on (ss.SchetSluchAccomplished=ssa.SchetSluchAccomplishedID)
	                where(
		            (s.[year] in (2018,2019,2020) and ssa.oplata=1 and s.type_=693 and isnull(ssa.usl_ok,3)<>4 and (s.FILENAME like 'HS%' or s.FILENAME like 'HCS%' or s.FILENAME like 'D_S%' or s.FILENAME like 'TS%')))
		            and datediff(month,sz.dr,ss.date_1)>=216
		            and (ss.ds1 like 'E10%' or ss.ds1 like 'E11%')

select sz.srz_pid, min(date_1) as date_1
into #disp
    from iesdb.[IES].T_SCHET_SLUCH as ss with(nolock)
	    inner join iesdb.ies.T_SCHET_ZAP as sz with(nolock) on (sz.SchetZapID = ss.SchetZap)
	        inner join iesdb.ies.T_SCHET as s with(nolock) on (s.SchetID=sz.Schet)
	            inner join iesdb.ies.T_SCHET_SLUCH_ACCOMPLISHED as ssa with(nolock) on (ss.SchetSluchAccomplished=ssa.SchetSluchAccomplishedID)
	                where s.[year]=@year and s.[MONTH]<=@month
	                and ssa.oplata=1 and s.type_=693 and s.Status=1 and (s.FILENAME like 'DPM%' or s.FILENAME like 'DOM%') and (ss.ds1 like 'E10%' or ss.ds1 like 'E11%')
	                and isnull(ssa.usl_ok,3)<>4
	                and datediff(month,sz.dr,ss.date_1)>=216
	                and not exists (select top 1 1 from #allsluch a where a.srz_pid=sz.srz_pid and a.DATE_1<ssa.DATE_Z_1 and a.SchetSluchID<>ss.SchetSluchID)
	                    group by srz_pid

select  sz.srz_pid
into #sluch
	from iesdb.[IES].T_SCHET_SLUCH as ss with(nolock)
    	inner join iesdb.ies.T_SCHET_ZAP as sz with(nolock) on (sz.SchetZapID = ss.SchetZap)
	        inner join iesdb.ies.T_SCHET as s with(nolock) on (s.SchetID=sz.Schet)
	            inner join iesdb.ies.T_SCHET_SLUCH_ACCOMPLISHED as ssa with(nolock) on (ss.SchetSluchAccomplished=ssa.SchetSluchAccomplishedID)
	                where s.[year]=@year and s.[MONTH]<=@month
	                and ssa.oplata=1 and s.type_=693 and s.Status=1 and
	                (s.FILENAME like 'HM%' or s.FILENAME like 'HCM%' or s.FILENAME like 'D_M%' or s.FILENAME like 'TM%')
	                and (ss.ds1 like 'E10%' or ss.ds1 like 'E11%')
	                and datediff(month,sz.dr,ss.date_1)>=216
	                and isnull(ssa.usl_ok,3)<>4
	                and not exists (select top 1 1 from #allsluchnew a where a.srz_pid=sz.srz_pid)
	                    group by srz_pid

select (select top 1 nam_mok from iesdb.IES.T_F003_MO t15  with(nolock) where t15.MCOD = convert(varchar(6),fj.lpu)) as nam_mok,count (distinct srz_pid) as col
	from #sluch d
    	inner join orange.[RPN_online].[dbo].[FinanceJournal] fj on fj.pid=d.srz_pid and fj.DTSECTION=convert(date,'2021-07-01',127)
	        group by fj.lpu
	            order by fj.lpu

drop table #allsluch,#disp,#sluch,#allsluchnew
